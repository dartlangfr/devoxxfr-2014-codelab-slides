lang-dartimport 'dart:math';

class Point {
    num x, y;
    Point(this.x, this.y);
    num distanceTo(Point other) {
        var dx = x - other.x;
        var dy = y - other.y;
        return sqrt(dx * dx + dy * dy);
    }
    String toString() => "($x, $y)"
}
